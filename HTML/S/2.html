  13 struct pcb *makepcb(struct pcbstate *pState, char *pszPCBName, char *pszPCBImage, enum pcbclass clas, int prior) {
  14         /* The PCB to return. */
  15         struct pcb *pPCB;
  16 
  17         /* Make sure priorities are valid. */
  18         if(prior < PCB_MINPRIOR || prior > PCB_MAXPRIOR) return NULL;
  19 
  20         /* Allocate a PCB and fail if allocation does. */
  21         pPCB = malloc(sizeof(struct pcb));
  22         assert(pPCB != NULL);
  23 
  24         /* Set PCB ID. */
  25         pPCB->id    = pState->nProcid++;
  26         /* Intern the string for the PCB name. */
  27         pPCB->kName = internstring(pState->ptPCBNames, pszPCBName);
  28 
  29         /* Initialize passed in properties. */
  30         pPCB->clas     = clas;
  31         pPCB->priority = prior;
  32 
  33         /* Initialize PCB status. */
  34         pPCB->status = PCB_READY;
  35         pPCB->susp   = PCB_FREE;
  36 
  37         /* Initialize PCB queue. */
  38         pPCB->pNext = pPCB;
  39         pPCB->pPrev = pPCB;
  40 
  41         /* Setup process data. */
  42         pPCB->kImage = internstring(pState->ptPCBImages, pszPCBImage);
  43         pPCB->offset = 0;
  44 
  45         return pPCB;

  54 static struct pcb *queuefindpcbnum(struct pcbqueue *pqQueue, int pcbnum) {
  55         switch(pqQueue->type) {
  56         case QT_NORMAL:
  57                 {
  58                         /* The number of the first PCB. */
  59                         int initPCB;
  60                         /* The current PCB to search. */
  61                         struct pcb *pPCB;
  62 
  63                         /* Initialize iteration. */
  64                         pPCB    = pqQueue->pHead;
  65                         initPCB = pPCB->id;
  66 
  67                         do {
  68                                 /* Return the PCB if it matches. */
  69                                 if(pPCB->id == pcbnum) return pPCB;
  70                                 /* Advance to the next PCB in the queue. */
  71                                 pPCB = pPCB->pNext;
  72                         } while(pPCB->id != initPCB);
  73 
  74                         return NULL;
  75                 }
  76         case QT_PRIORITY:
  77                 {
  78                         /* Loop index for queues. */
  79                         int i;
  80                         /* Cast queue. */
  81                         struct pcbqueueprior *pqrQueue;
  82 
  83                         pqrQueue = (struct pcbqueueprior *)pqQueue;
  84                         for(i = 0; i <= PCB_MAXPRIOR; i++) {
  85                                 /* The PCB we possibly found. */
  86                                 struct pcb *pPCB;
  87 
  88                                 /* Find in each priority level. */
  89                                 pPCB = queuefindpcbnum(pqrQueue->apqQueues[i], pcbnum);
  90                                 if(pPCB != NULL) return pPCB;
  91                         }
  92                         
  93                         /* We found nothing. */
  94                         return NULL;
  95                 }
  96         default:
  97                 /* Shouldn't happen. */
  98                 assert(0);
  99         }

 156 /* Find a PCB by number in all of the queues. */
 157 struct pcb *findpcbnum(struct pcbstate *pState, int pcbnum) {
 158         /* The PCB to return. */
 159         struct pcb *pPCB;
 160 
 161         /* Fail fast for numbers that don't exist. */
 162         if(pState->nProcid < pcbnum) return NULL;
 163 
 164         /* Search the queues in order. */
 165         pPCB                  = queuefindpcbnum(pState->pqReady,    pcbnum);
 166         if(pPCB == NULL) pPCB = queuefindpcbnum(pState->pqBlocked,  pcbnum);
 167         if(pPCB == NULL) pPCB = queuefindpcbnum(pState->pqsReady,   pcbnum);
 168         if(pPCB == NULL) pPCB = queuefindpcbnum(pState->pqsBlocked, pcbnum);
 169 
 170         return pPCB;
 171 }

 210 /* Insert a PCB into a FIFO queue. */
 211 static void fifoinsertpcb(struct pcbqueue *pqQueue, struct pcb *pPCB) {
 212         /* Only valid on normal queues. */
 213         assert(pqQueue->type == QT_NORMAL);
 214 
 215         /* The current head of the chain. */
 216         struct pcb *pHead;
 217         /* The current tail of the chain. */
 218         struct pcb *pTail;
 219 
 220         /* Fill an queue if it is empty. */
 221         if(fillqueue(pqQueue, pPCB)) return;
 222 
 223         pHead = pqQueue->pHead;
 224         pTail = pHead->pPrev;
 225 
 226         /* Chain the PCB to the back. */
 227         pPCB->pNext = pHead;
 228         pPCB->pPrev = pTail;
 229 
 230         /* Adjust the queue members. */
 231         pTail->pNext = pPCB;
 232         pHead->pPrev = pPCB;
 233 }

 278 /* Insert a PCB into the proper queue. */
 279 enum pcberror insertpcb(struct pcbstate *pState, struct pcb *pPCB) {
 280         /* The two queues to select from. */
 281         struct pcbqueue *pqReady;
 282         struct pcbqueue *pqBlocked;
 283 
 284         /* Don't insert a PCB that is already in a queue. */
 285         if(pPCB->pNext != pPCB || pPCB->pPrev != pPCB) return PCBINQUEUE;
 286 
 287         switch(pPCB->susp) {
 288         case PCB_FREE:
 289                 pqReady   = pState->pqReady;
 290                 pqBlocked = pState->pqBlocked;
 291                 break;
 292         case PCB_SUSPENDED:
 293                 pqReady   = pState->pqsReady;
 294                 pqBlocked = pState->pqsBlocked;
 295                 break;
 296         default:
 297                 return PCBINVSUSP;
 298         }
 299 
 300         switch(pPCB->status) {
 301         case PCB_READY:
 302                 doinsertpcb(pqReady, pPCB);
 303                 pqReady->nprocs += 1;
 304                 break;
 305         case PCB_BLOCKED:
 306                 doinsertpcb(pqBlocked, pPCB);
 307                 pqBlocked->nprocs += 1;
 308                 break;
 309         case PCB_RUNNING:
 310                 return PCBRUNNING;
 311         default:
 312                 return PCBINVSTAT;
 313         }
 314 
 315         return 0;
 316 }

 318 /* Remove a PCB from a queue. */
 319 void removepcb(struct pcbstate *pState, struct pcb *pPCB) {
 320         /* Temporary links that will be restored. */
 321         struct pcb *pPrev, *pNext;
 322         /* The queue this PCB came from. */
 323         struct pcbqueue *pqQueue;
 324 
 325         if(pPCB->susp == PCB_SUSPENDED && pPCB->status == PCB_READY) {
 326                 pqQueue = pState->pqsReady;
 327         } else if(pPCB->susp == PCB_SUSPENDED && pPCB->status == PCB_BLOCKED) {
 328                 pqQueue = pState->pqsBlocked;
 329         } else if(pPCB->susp == PCB_FREE && pPCB->status == PCB_READY) {
 330                 pqQueue = pState->pqReady;
 331         } else if(pPCB->susp == PCB_FREE && pPCB->status == PCB_BLOCKED) {
 332                 pqQueue = pState->pqBlocked;
 333         } else {
 334                 printf("BANG! Attempted to remove a PCB not in a queue.\n");
 335                 assert(0);
 336         }
 337 
 338         /* Save the links to restore. */
 339         pNext = pPCB->pNext;
 340         pPrev = pPCB->pPrev;
 341 
 342         /* Remove the PCB from the queue. */
 343         pPCB->pNext = pPCB;
 344         pPCB->pPrev = pPCB;
 345 
 346         /* Set the proper queue. */
 347         switch(pqQueue->type) {
 348         case QT_NORMAL:
 349                 break;
 350         case QT_PRIORITY:
 351                 pqQueue->nprocs -= 1;
 352                 pqQueue = ((struct pcbqueueprior *)pqQueue)->apqQueues[pPCB->priority];
 353                 break;
 354         default:
 355                 /* Shouldn't happen. */
 356                 assert(0);
 357         }
 358 
 359         pqQueue->nprocs -= 1;
 360 
 361         if(pqQueue->nprocs < 1) {
 362                 /* Empty a queue. */
 363                 pqQueue->pHead = NULL;
 364         } else {
 365                 /* Restore the queue links. */
 366                 pNext->pPrev = pPrev;
 367                 pPrev->pNext = pNext;
 368 
 369                 if(pqQueue->pHead == pPCB) {
 370                         /* Reset queue head. */
 371                         pqQueue->pHead = pNext;
 372                 }
 373         }
 374 }

 376 /* Print a PCB. */
 377 void printpcb(struct pcb *pPCB, void *pvState) {
 378         /* The OS state. */
 379         struct osstate *ostate;
 380 
 381         /* The name of the PCB. */
 382         const char *pszPCBName;
 383         /* The process image for the PCB. */
 384         const char *pszPCBImage;
 385         /* The class of the PCB. */
 386         const char *pszPCBClass;
 387         /* The status of the PCB. */
 388         const char *pszPCBStatus;
 389         /* The suspension status of the PCB. */
 390         const char *pszPCBSusp;
 391 
 392         /* Cast arg. */
 393         ostate = (struct osstate *)(pvState);
 394 
 395         /* Get the PCB name. */
 396         pszPCBName  = lookupkey(ostate->pPCBstat->ptPCBNames, pPCB->kName);
 397         /* Get the image name. */
 398         pszPCBImage = lookupkey(ostate->pPCBstat->ptPCBImages, pPCB->kImage);
 399 
 400         /* Get the PCB class. */
 401         switch(pPCB->clas) {
 402         case PCB_SYSTEM:
 403                 pszPCBClass = "System";
 404                 break;
 405         case PCB_APPLICATION:
 406                 pszPCBClass = "Application";
 407                 break;
 408         default:
 409                 /* Shouldn't happen. */
 410                 assert(0);
 411         }
 412 
 413         /* Get the PCB status. */
 414         switch(pPCB->status) {
 415         case PCB_READY:
 416                 pszPCBStatus = "Ready";
 417                 break;
 418         case PCB_RUNNING:
 419                 pszPCBStatus = "Running";
 420                 break;
 421         case PCB_BLOCKED:
 422                 pszPCBStatus = "Blocked";
 423                 break;
 424         default:
 425                 /* Shouldn't happen. */
 426                 assert(0);
 427         }
 428 
 429         /* Get the PCB suspension status. */
 430         switch(pPCB->susp) {
 431         case PCB_SUSPENDED:
 432                 pszPCBSusp = "Yes";
 433                 break;
 434         case PCB_FREE:
 435                 pszPCBSusp = "No";
 436                 break;
 437         default:
 438                 /* Shouldn't happen. */
 439                 assert(0);
 440         }
 441 
 442         /* Print basic PCB information. */
 443         fprintf(ostate->output, "PCB ID:           %d\n", pPCB->id);
 444         fprintf(ostate->output, "PCB Name:         %s\n", pszPCBName);
 445         fprintf(ostate->output, "PCB Class:        %s\n", pszPCBClass);
 446         fprintf(ostate->output, "PCB Priority:     %d\n", pPCB->priority);
 447 
 448         /* Print PCB status information. */
 449         fprintf(ostate->output, "PCB Status:       %s\n", pszPCBStatus);
 450         fprintf(ostate->output, "Is PCB Suspended: %s\n", pszPCBSusp);
 451 
 452         /* Print PCB process information. */
 453         fprintf(ostate->output, "PCB Image:         %s\n", pszPCBImage);
 454         fprintf(ostate->output, "PCB Image Offset:  %d\n\n", pPCB->offset);
 455 }

 457 /* Check if there are any processes to dispatch. */
 458 int candispatch(struct pcbstate *pState) {
 459         if(pState->pqReady->nprocs > 0 || pState->pqBlocked->nprocs > 0) {
 460                 return pState->pqReady->nprocs + pState->pqBlocked->nprocs;
 461         }
 462 
 463         return 0;
 464 }

 466 /* Get a dispatchable process. */
 467 struct pcb *getdisppcb(struct pcbstate *pState) {
 468         if(pState->pqReady->nprocs > 0 && pState->pqBlocked->nprocs > 0) {
 469                 if(random() > .5) return poppcb(pState, pState->pqReady);
 470                 else              return poppcb(pState, pState->pqBlocked);
 471         } else if(pState->pqReady->nprocs > 0) {
 472                 return poppcb(pState, pState->pqReady);
 473         } else if(pState->pqBlocked->nprocs > 0) {
 474                 return poppcb(pState, pState->pqBlocked);
 475         } else {
 476                 return NULL;
 477         }

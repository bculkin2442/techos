  21 /* Convert a string to a hashcode. */
  22 static unsigned long hashstring(const char *val) {
  23         /* This is the DJB2K33 algorithm. It generally works quite well. */
  24         unsigned long hash = 5381;
  25         int c;
  26 
  27         /* Get the current string character. */
  28         c    = *val;
  29         val += 1;
  30 
  31         /* While we're not at the end of the string. */
  32         while(c) {
  33                 /* hash * 33 + c */
  34                 hash = ((hash << 5) + hash) + c; 
  35 
  36                 /* Advance to the next character. */
  37                 c    = *val;
  38                 val += 1;
  39         }
  40 
  41         return hash;
  42 }

 109 /* Destroy a bucket chain. */
 110 static void killbucket(struct bucket *buck, int freestrings) {
 111         /* Terminate the chain. */
 112         /*
 113          * NOTE:
 114          *      buck->prev is only null in the case where nothing has
 115          *      been inserted into this bucket before. 
 116          */
 117         if(buck->prev != NULL) {
 118                 buck->prev->next = NULL;
 119         }
 120 
 121         while(buck != NULL && buck->next != NULL) {
 122                 /* Next bucket. */
 123                 struct bucket *tmp;
 124 
 125                 /* Advance the queue. */
 126                 tmp  = buck;
 127                 buck = buck->next;
 128 
 129                 /* Free the string we own. */
 130                 if(tmp->val != NULL && freestrings == 1) free(tmp->val);
 131                 /* Free the bucket. */
 132                 free(tmp);
 133         }
 134 }

 152 /* Add a bucket to a bucket chain. */
 153 static void addbucket(struct bucket *bucket) {
 154         /* Allocate a new bucket. */
 155         struct bucket *nbucket;
 156         nbucket = malloc(sizeof(struct bucket));
 157         assert(nbucket != NULL);
 158 
 159         if(bucket->next == NULL) {
 160                 /* Solo bucket. */
 161                 bucket->next  = nbucket;
 162                 bucket->prev  = nbucket;
 163 
 164                 nbucket->next = bucket;
 165                 nbucket->prev = bucket;
 166         } else {
 167                 /* Chain bucket. */
 168 
 169                 /* The next bucket in the list. */
 170                 struct bucket *nextbucket;
 171                 nextbucket = bucket->next;
 172 
 173                 /* Insert our bucket into the chain. */
 174                 bucket->next     = nbucket;
 175                 nextbucket->prev = nbucket;
 176 
 177                 nbucket->next = nextbucket;
 178                 nbucket->prev = bucket;
 179         }

 182 /* Intern a string. */
 183 internkey internstring(struct interntab *table, const char *string) {
 184         /* The intern key we use. */
 185         internkey key;
 186         /* The key/string hashes. */
 187         int khash, shash;
 188         /* The string/key buckets. */
 189         struct bucket *sbucket, *kbucket;
 190 
 191         /* Don't duplicate an already interned string. */
 192         key = lookupstring(table, string);
 193         if(key != SIINVALID) return key;
 194 
 195         /* Get the next valid intern key. */
 196         key = table->nextkey;
 197         table->nextkey += 1;
 198         /* Fail if we ran out of space for strings. */
 199         if(key < 0) return SITABFULL;
 200         /* Make sure we never make the invalid key valid. */
 201         assert(key != SIINVALID);
 202 
 203         /* Create a copy of the string we own. */
 204         char *strang = (char *)strdup(string);
 205 
 206         /* Get both hashes. */
 207         shash = hashstring(strang) % BUCKET_COUNT;
 208         khash = hashkey(key)       % BUCKET_COUNT;
 209 
 210         /* Get the buckets for each hash. */
 211         sbucket = table->strings[shash];
 212         kbucket = table->keys[khash];
 213 
 214         if(sbucket->key != SIINVALID) {
 215                 /*
 216                  * Non-empty bucket. 
 217                  *
 218                  * Insert into a new one.
 219                  */
 220                 sbucket = sbucket->prev;
 221         }
 222         if(kbucket->key != SIINVALID) {
 223                 /*
 224                  * Non-empty bucket. 
 225                  *
 226                  * Insert into a new one.
 227                  */
 228                 kbucket = kbucket->prev;
 229         }
 230 
 231         /* Insert into the right bucket in both tables. */
 232         sbucket->val = strang;
 233         kbucket->val = strang;
 234 
 235         sbucket->key = key;
 236         kbucket->key = key;
 237 
 238         /* Add a new empty bucket to both tables. */
 239         addbucket(sbucket);
 240         addbucket(kbucket);
 241 
 242         return key;
 243 }

 245 /* Lookup a interned key for a string. */
 246 internkey lookupstring(struct interntab *table, const char *string) {
 247         /* The string hash. */
 248         int shash;
 249         /* The string bucket. */
 250         struct bucket *sbucket;
 251         /* The first key in a bucket. */
 252         int fkey;
 253 
 254         /* Get the bucket for this string. */
 255         shash   = hashstring(string) % BUCKET_COUNT;
 256         sbucket = table->strings[shash];
 257         fkey    = sbucket->key;
 258 
 259         /* Bail out early if the bucket is empty. */
 260         if(sbucket->key == SIINVALID) return SIINVALID;
 261 
 262         do {
 263                 /* Only check buckets with valid values. */
 264                 if(sbucket->val != NULL) {
 265                         /* Check if this is the right bucket. */
 266                         if(strcmp(string, sbucket->val) == 0) return sbucket->key;
 267                 }
 268 
 269                 /* Move to next bucket. */
 270                 sbucket = sbucket->next;
 271 
 272                 /* Bail out if we've looped or hit an empty bucket. */
 273         } while(sbucket->key != fkey && sbucket->key != SIINVALID);
 274 
 275         /* We didn't find anything. */
 276         return SIINVALID;
 277 }

 279 /* Lookup the string for an interned key. */
 280 const char *lookupkey(struct interntab *table, const internkey key) {
 281         /* The key hash. */
 282         int khash;
 283         /* The key bucket. */
 284         struct bucket *kbucket;
 285         /* The first key in a bucket. */
 286         int fkey;
 287 
 288         /* This key can't possibly be in the table. */
 289         if(table->nextkey < key) return NULL;
 290 
 291         /* Get the bucket for this key. */
 292         khash   = hashkey(key) % BUCKET_COUNT;
 293         kbucket = table->keys[khash];
 294         fkey    = kbucket->key;
 295 
 296         /* Bail out early if the bucket is empty. */
 297         if(fkey == SIINVALID) return NULL;
 298 
 299         do {
 300                 /* Check if this is the right bucket. */
 301                 if(key == kbucket->key) return kbucket->val;
 302 
 303                 /* Move to next bucket. */
 304                 kbucket = kbucket->next;
 305 
 306                 /* Bail out if we've looped or hit an empty bucket. */
 307         } while(kbucket->key != fkey && kbucket->key != SIINVALID);
 308 
 309         /*
 310          * We didn't find anything. */
 311         return NULL;
 312 }

 314 /* Execute an action for every key in a table. */
 315 void foreachintern(struct interntab *table, tableitr itr, void *pvData) {
 316         int i;
 317         /* Iterate over every bucket. */
 318         for(i = 0; i < BUCKET_COUNT; i++) {
 319                 struct bucket *buck;
 320                 internkey      kFirst;
 321                 
 322                 /* Get the first bucket, and its key. */
 323                 buck = table->keys[i];
 324                 kFirst = buck->key;
 325                 
 326                 /* Ignore this bucket if its empty. */
 327                 if(kFirst == SIINVALID) continue;
 328 
 329                 do {
 330                         itr(buck->val, buck->key, pvData);
 331                         /* Bail out if we've looped or hit an empty bucket. */
 332                 } while(buck->key != kFirst && buck->key != SIINVALID);
 333         }
 334 }

 395 /* Insert an item into the list. */
 396 void putinternlist(struct internlist *plList, char *pszKey, void *pvData) {
 397         /* Intern the name, and fail if interning fails */
 398         internkey  kData = internstring(plList->ptKeys, pszKey);
 399         assert(kData > 0);
 400 
 401         if(plList->dataspace == plList->datausage) {
 402                 /* Allocate more space for the list. */
 403                 plList->dataspace *= 2;
 404                 plList->dataspace += 1;
 405                 plList->paData     = realloc(plList->paData, sizeof(void *) * plList->dataspace);
 406                 /* Fail if allocation fails. */
 407                 assert(plList->paData != NULL);
 408         }
 409 
 410         /* Delete any item that was in that slot. */
 411         if(plList->paData[kData -1] != NULL) {
 412                 plList->pfDestroy(plList->paData[kData - 1]);
 413         }
 414 
 415         /* Insert the item. */
 416         plList->paData[kData - 1]  = pvData;
 417         plList->datausage         += 1;
 418 }

 425 void *getinternlist(struct internlist * plList, char *pszKey) {
 426         /* Lookup the index. */
 427         internkey kData = lookupstring(plList->ptKeys, pszKey);
 428         /* Error if we don't have that key. */

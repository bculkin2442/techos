  27 HANDLECOM(mkpcb) { 
  28         /* Class of the PCB. */
  29         enum pcbclass clas;
  30         /* Priority of the PCB. */
  31         int priority;
  32         /* Name of the PCB. */
  33         char *pszPCBName;
  34 
  35         /* Current option/long option. */
  36         int opt, optidx;
  37 
  38         /* Set default options. */
  39         clas       = PCB_APPLICATION;
  40         pszPCBName = NULL;
  41 
  42         /* Reinit getopt. */
  43         optind = 1;
  44 
  45         /* Process options. */
  46         while(1) {
  47                 /* Usage message. */
  48                 char *usage = "Usage: mkpcb [-cnh] [-c|--class (app|application)|(system|sys)] [--help] [-n|--name <proc-name>] <image-name> <priority>\n";
  49 
  50                 /* The long options we take. */
  51                 static struct option opts[] = {
  52                         /*class options*/
  53                         {"class", required_argument, 0, 'c'},
  54 
  55                         /* Misc. options. */
  56                         {"help", no_argument,       0, 'h'},
  57                         {"name", required_argument, 0, 'n'},
  58 
  59                         /* Terminating option. */
  60                         {0, 0, 0, 0}
  61                 };
  62 
  63                 /* Get an option. */
  64                 opt = getopt_long(argc, argv, "c:n:h", opts, &optidx);
  65                 
  66                 /* Break if we've processed every option. */
  67                 if(opt == -1) break;
  68 
  69                 /* Handle options. */
  70                 switch(opt) {
  71                 case 0:
  72                         /*
  73                          * Long-options with corrsponding short-options are
  74                          * handled by the corresponding short option. 
  75                          */
  76                         switch(optidx) {
  77                         default:
  78                                 fprintf(ostate->output, "\tERROR: Invalid command-line argument\n");
  79                                 fprintf(ostate->output, "%s\n", usage);
  80                                 return 1;
  81                         }
  82                         break;
  83                 case 'c':
  84                         if(strcmp(optarg, "application") == 0 ||
  85                            strcmp(optarg, "app")         == 0) {
  86                                 clas = PCB_APPLICATION;
  87                         } else if(strcmp(optarg, "system") == 0 ||
  88                                   strcmp(optarg, "sys")    == 0) {
  89                                 clas = PCB_SYSTEM;
  90                         } else {
  91                                 fprintf(ostate->output, "ERROR: Invalid PCB class '%s' (valid classes are 'app'/'application' and 'sys/system')\n", optarg);
  92 
  93                                 return 1;
  94                         }
  95                         break;
  96                 case 'h':
  97                         fprintf(ostate->output, "%s\n", usage);
  98                         return 0;
  99                 case 'n':
 100                         /* Free an already set name. */
 101                         if(pszPCBName != NULL) {
 102                                 free(pszPCBName);
 103                         }
 104                         /* 
 105                          * @NOTE
 106                          *      Do we need to duplicate this?
 107                          */
 108                         pszPCBName = (char *)strdup(optarg);
 109                         assert(pszPCBName != NULL);
 110                         break;
 111                 default:
 112                         fprintf(ostate->output, "\tERROR: Invalid command-line argument.\n");
 113                         fprintf(ostate->output, "%s\n", usage);
 114                         return 1;
 115                 }
 116 
 117         }
 118 
 119         /* Check if we have a priority as the second non-arg option. */
 120         if(argc >= (optind + 2)){
 121                 /* Priority argument. */
 122                 char *prarg;
 123                 prarg = argv[optind + 1];
 124 
 125                 /* Parse priority. */
 126                 if(sscanf(prarg, "%d", &priority) < 1) {
 127                         fprintf(ostate->output, "ERROR: '%s' is not a valid priority (must be an integer)\n", prarg);
 128                         return 1;
 129                 }
 130         } else {
 131                 fprintf(ostate->output, "ERROR: Priority not given.\n");
 132                 return 1;
 133         }
 134 
 135         /* Check the PCB priority is valid. */
 136         if(priority < PCB_MINPRIOR || priority > PCB_MAXPRIOR) {
 137                 fprintf(ostate->output, "ERROR: Priority entered is out of bounds (must be between %d and %d)\n", PCB_MINPRIOR, PCB_MAXPRIOR);
 138                 return 1;
 139         }
 140 
 141         /* Bind the PCB image. */
 142         char *pszPCBImage = argv[optind];
 143 
 144         /* Validate that the image referenced exists. */
 145         {
 146                 struct stat scratch;
 147 
 148                 /* Error if there isn't a normal file for the image. */
 149                 if(stat(pszPCBImage, &scratch) == -1) {
 150                         fprintf(ostate->output, "ERROR: No image found for process '%s'\n", pszPCBImage);
 151                         return 1;
 152                 } else if(!S_ISREG(scratch.st_mode)) {
 153                         fprintf(ostate->output, "ERROR: Invalid image for process '%s'\n", pszPCBImage);
 154                         return 1;
 155                 }
 156         }
 157 
 158         /* Set the PCB name correctly. */
 159         if(pszPCBName == NULL) {
 160                 pszPCBName = pszPCBImage;
 161         }
 162

 164         struct pcb *madePCB = makepcb(ostate->pPCBstat, pszPCBName, pszPCBImage, clas, priority);
 165         if(madePCB == NULL) {
 166                 fprintf(ostate->output, "INTERNAL ERROR: PCB named '%s' created as null.\n", pszPCBImage);
 167                 return 1;
 168         }
 169 
 170         {
 171                 /* Return from insertpcb. */
 172                 enum pcberror pcbstat;
 173 
 174                 pcbstat = insertpcb(ostate->pPCBstat, madePCB);

 176                 switch(pcbstat) {

 183                         fprintf(ostate->output, "INTERNAL ERROR: New PCB %d (named '%s') has invalid run state %d\n",
 184                                         madePCB->id, pszPCBImage, madePCB->status);
 185                         break;
 186                 case PCBRUNNING:
 187                         fprintf(ostate->output, "INTERNAL ERROR: New PCB %d (named '%s') is already running\n",
 188                                         madePCB->id, pszPCBImage);
 189                         break;
 190                 case PCBINQUEUE:
 191                         fprintf(ostate->output, "INTERNAL ERROR: New PCB %d (named '%s') is already in a queue\n",
 192                                         madePCB->id, pszPCBImage);
 193                         break;
 194                 default:
 195                         fprintf(ostate->output, "INTERNAL ERROR: Unknown return %d from attempting to insert new PCB %d (named '%s)\n", pcbstat, madePCB->id, pszPCBImage);
 196                 }
 197 
 198                 return 1;
 199         }

 719         enum queueopt {
 720                 /* Show the ready queues. */
 721                 QU_READY,
 722                 /* Show the blocked queues. */
 723                 QU_BLOCKED,
 724                 /* Show all the queues. */
 725                 QU_ALL
 726         };
 727         /* Process ID options. */
 728         enum pidopt {
 729                 /* Locate a PCB by name. */
 730                 PID_NAME,
 731                 /* Locate a PCB by number. */
 732                 PID_NUM
 733         };
 734 
 735         /* Chosen mode. */
 736         enum showopt  mode   = SHOW_QUEUE;
 737         /* Chosen queue. */
 738         enum queueopt queue  = QU_ALL;
 739         /* Chosen ID method. */
 740         enum pidopt   idtype = PID_NAME;
 741 
 742         /* Reinit getopt. */
 743         optind = 1;
 744 
 745         /* Parse CLI args. */
 746         while(1) {
 747                 /* Enum declaration for long options. */
 748                 enum shpcbopt {
 749                         /* Help option. */
 750                         SH_HELP = 0,
 751 
 752                         /* Mode options. */
 753                         SH_MODE,       /* Specify the command mode. */
 754                         SH_QUEUE,      /* Specify the queue to show. */
 755                         SH_PROC,       /* Specify the way to locate a process. */
 756 
 757                 };
 758 
 759                 /* The current option, and the current long option. */
 760                 int opt, optidx;
 761 
 762                 /* Our usage message. */
 763                 static const char *usage = "Usage: shpcb [-h] [--help] [--mode pcb|queue] [--queue ready|blocked|all] [--proc name|num] <proc-name>|<proc-id>\n";
 764 
 765 
 766                 /* The long options we take. */
 767                 /* 
 768                  * :OptHandling
 769                  */
 770                 static struct option opts[] = {
 771                         /* Misc. options. */
 772                         {"help", no_argument, 0, 0},
 773 
 774                         /* Mode options. */
 775                         {"mode",  required_argument, 0, 0},
 776                         {"queue", required_argument, 0, 0},
 777                         {"proc",  required_argument, 0, 0},
 778 
 779                         /* Terminating argument. */
 780                         {0, 0, 0, 0}
 781                 };
 782 
 783                 /* Get an option. */
 784                 opt = getopt_long(argc, argv, "h", opts, &optidx);
 785                 /* Break if there are no more options. */
 786                 if(opt == -1) break;
 787 
 788                 /* Handle options. */
 789                 /* :OptHandling */
 790                 switch(opt) {
 791                 case 0:
 792                         /* Handle long options. */
 793                         switch(optidx) {
 794                         case SH_HELP:
 795                                 fprintf(ostate->output, "%s\n", usage);
 796                                 return 0;
 797                         case SH_MODE:
 798                                 if(strcmp(optarg, "pcb") == 0) {
 799                                         mode = SHOW_PCB;
 800                                 } else if(strcmp(optarg, "queue") == 0) {
 801                                         mode = SHOW_QUEUE;
 802                                 } else {
 803                                         fprintf(ostate->output, "ERROR: Invalid mode '%s'. Valid modes are 'pcb' and 'queue'\n", optarg);
 804                                         return 1;
 805                                 }
 806                                 break;
 807                         case SH_QUEUE:
 808                                 if(strcmp(optarg, "ready") == 0) {
 809                                         queue = QU_READY;
 810                                 } else if(strcmp(optarg, "blocked") == 0) {
 811                                         queue = QU_BLOCKED;
 812                                 } else if(strcmp(optarg, "all") == 0) {
 813                                         queue = QU_ALL;
 814                                 } else {
 815                                         fprintf(ostate->output, "ERROR: Invalid queue '%s'. Valid queues are 'ready', 'blocked' and 'all'\n", optarg);
 816                                         return 1;
 817                                 }
 818                                 break;
 819                         case SH_PROC:
 820                                 if(strcmp(optarg, "name") == 0) {
 821                                         idtype = PID_NAME;
 822                                 } else if(strcmp(optarg, "num") == 0) {
 823                                         idtype = PID_NUM;
 824                                 } else {
 825                                         fprintf(ostate->output, "ERROR: Invalid process ID type '%s'. Valid ID types are 'name' and 'num'\n", optarg);
 826                                         return 1;
 827                                 }
 828                                 break;
 829                         default:
 830                                 fprintf(ostate->output, "\tERROR: Invalid command-line argument\n");
 831                                 fprintf(ostate->output, "%s\n", usage);
 832                                 return 1;
 833                         }
 834                         break;
 835                 case 'h':
 836                         fprintf(ostate->output, "%s\n", usage);
 837                         return 0;
 838                 default:
 839                         fprintf(ostate->output, "\tERROR: Invalid command-line argument\n");
 840                         fprintf(ostate->output, "%s\n", usage);
 841                 }
 842         }
 843 
 844         switch(mode) {
 845         case SHOW_PCB:
 846                 {
 847                         /* The PCB to show. */
 848                         struct pcb *pPCB;
 849 
 850                         switch(idtype) {
 851                         case PID_NAME:
 852                                 if(optind < argc) {
 853                                         char *pszPCBName = argv[optind];
 854                                         pPCB = findpcbname(ostate->pPCBstat, pszPCBName);
 855                                         if(pPCB == NULL) {
 856                                                 fprintf(ostate->output, "ERROR: No PCB with name '%s'\n", pszPCBName);
 857                                                 return 1;
 858                                         }
 859                                 } else {
 860                                         fprintf(ostate->output, "ERROR: Must specify PCB name as argument.\n");
 861                                         return 1;
 862                                 }
 863 
 864                                 break;
 865                         case PID_NUM:
 866                                 if(optind < argc) {
 867                                         int pcbid = atoi(argv[optind]);
 868                                         pPCB = findpcbnum(ostate->pPCBstat, pcbid);
 869                                         if(pPCB == NULL) {
 870                                                 fprintf(ostate->output, "ERROR: No PCB with ID '%d'\n", pcbid);
 871                                                 return 1;
 872                                         }
 873                                 } else {
 874                                         fprintf(ostate->output, "ERROR: Must specify PCB name as argument.\n");
 875                                         return 1;
 876                                 }
 877                                 break;
 878                         default:
 879                                 /* Shouldn't happen. */
 880                                 assert(0);
 881                         }
 882                         printpcb(pPCB, ostate);
 883                 }
 884                 break;
 885         case SHOW_QUEUE:
 886                 switch(queue) {
 887                 case QU_READY:
 888                         {
 889                                 /* No. of processes. */
 890                                 int nprocs;
 891 
 892                                 nprocs = 0                                +
 893                                         ostate->pPCBstat->pqReady->nprocs +
 894                                         ostate->pPCBstat->pqsReady->nprocs;
 895 
 896                                 fprintf(ostate->output, "No. of Ready Processes: %d\n", nprocs);
 897                                 foreachpcb(ostate->pPCBstat->pqReady,
 898                                                 &printpcb, ostate);
 899                                 foreachpcb(ostate->pPCBstat->pqsReady,
 900                                                 &printpcb, ostate);
 901                         }
 902                         break;
 903                 case QU_BLOCKED:
 904                         {
 905                                 /* No. of processes. */
 906                                 int nprocs;
 907 
 908                                 nprocs = 0                                  +
 909                                         ostate->pPCBstat->pqBlocked->nprocs +
 910                                         ostate->pPCBstat->pqsBlocked->nprocs;
 911 
 912                                 fprintf(ostate->output, "No. of Blocked Processes: %d\n", nprocs);
 913                                 foreachpcb(ostate->pPCBstat->pqBlocked,
 914                                                 &printpcb, ostate);
 915                                 foreachpcb(ostate->pPCBstat->pqsBlocked,
 916                                                 &printpcb, ostate);
 917                         }
 918                         break;
 919                 case QU_ALL:
 920                         {
 921                                 /* No. of processes. */
 922                                 int nprocs;
 923                                 nprocs = 0                                   + 
 924                                         ostate->pPCBstat->pqReady->nprocs    +
 925                                         ostate->pPCBstat->pqBlocked->nprocs  +
 926                                         ostate->pPCBstat->pqsReady->nprocs   + 
 927                                         ostate->pPCBstat->pqsBlocked->nprocs;
 928 
 929                                 fprintf(ostate->output, "No. of Processes: %d\n", nprocs);
 930                                 fprintf(ostate->output, "\nReady Processes:\n");
 931                                 foreachpcb(ostate->pPCBstat->pqReady,
 932                                                 &printpcb, ostate);
 933                                 fprintf(ostate->output, "\nBlocked Processes:\n");
 934                                 foreachpcb(ostate->pPCBstat->pqBlocked,
 935                                                 &printpcb, ostate);
 936                                 fprintf(ostate->output, "\nSuspended Ready Processes:\n");
 937                                 foreachpcb(ostate->pPCBstat->pqsReady,
 938                                                 &printpcb, ostate);
 939                                 fprintf(ostate->output, "\nSuspended Suspended Processes:\n");
 940                                 foreachpcb(ostate->pPCBstat->pqsBlocked,
 941                                                 &printpcb, ostate);
 942                         }
 943                         break;
 944                 default:
 945                         /* Shouldn't happen. */
 946                         assert(0);
 947                 }
 948                 break;
 949         default:
 950                 /* Shouldn't happen. */
 951                 assert(0);
 952         }
 953         return 0;
 954 }

  81 /*
  82  * Handle exiting from the prompt.
  83  *
  84  * Make sure to confirm before hand.
  85  */
  86 HANDLECOM(exit) {
  87         /* Variables for line input. */
  88         char  *pszLine = NULL;
  89         size_t lread, lsize;
  90 
  91         /* Command return status. */
  92         int ret;
  93 
  94         /* Init vars. */
  95         ret     = 0;
  96         pszLine = NULL;
  97 
  98         /* Handle CLI args. */
  99         if(argc > 1) {
 100                 return checkhelpargs(argc, argv, "Usage: exit [-h] [--help]\n", ostate);
 101         }
 102 
 103         fprintf(ostate->output, "Are you sure you want to exit? (y/n) ");
 104         lread = getline(&pszLine, &lsize, stdin);
 105 
 106         /* Use rpmatch() to pick an action. */
 107         if(lread > 1) {
 108                 int result = rpmatch(pszLine);
 109 
 110                 switch(result) {
 111                 case 1:
 112                         ret = -1;
 113                         break;
 114                 case 0:
 115                         ret = 0;
 116                         break;
 117                 case -1:
 118                 default:
 119                         fprintf(ostate->output, "ERROR: Unknown response '%s'\n", pszLine);
 120                 }
 121         }
 122 
 123         /*
 124          * Cleanup after ourselves.
 125          */
 126         if(pszLine != NULL) free(pszLine);
 127 
 128         return ret;
 129 }

 132 HANDLECOM(version) {
 133                 return checkhelpargs(argc, argv, "Usage: version [-advh] [--author] [--date] [--version] [--help]\n", ostate);
 134 
 135         /* Should version info be printed? */
 136         int printVersion = 1;
 137         /* Should author information be printed? */
 138         int printAuthor  = 0;
 139         /* Should date information be printed? */
 140         int printDate    = 0;
 141 
 142         /* Reinit getopt. */
 143         optind = 1;
 144 
 145         /* Process options. */
 146         while(1) {
 147                 /*
 148                  * @TODO 10/07/17 Ben Culkin :VersionOpts
 149                  *      Adjust these options to some better way of indicating
 150                  *      which information we do/do not want to show. If this was
 151                  *      java, I'd just go with a custom format string sorta
 152                  *      thing, but that'd have to be reimplemented for C, and'd
 153                  *      be kinda painful. Maybe we'd want to look into something
 154                  *      with getsubopt()?
 155                  */
 156 
 157                 /* The current option/long option. */
 158                 int opt, optidx;
 159                 /* Usage message. */
 160                 static char *usage = "Usage: version [-advh] [--author] [--date] [--version] [--help]\n";
 161 
 162                 static struct option longopts[] = {
 163                         /* Info ctl. options. */
 164                         {"author",  no_argument, 0, 'a'},
 165                         {"date",    no_argument, 0, 'd'},
 166                         {"version", no_argument, 0, 'v'},
 167 
 168                         /* Misc. options. */
 169                         {"help", no_argument, 0, 'h'},
 170 
 171                         /* Terminating option. */
 172                         {0, 0, 0, 0}
 173                 };
 174 
 175                 /* Get an option. */
 176                 opt = getopt_long(argc, argv, "advh", longopts, &optidx);
 177 
 178                 /* Finish options. */
 179                 if(opt == -1) break;
 180 
 181                 switch(opt) {
 182                         /* a, d and v are toggle options. */
 183                 case 'a':
 184                         printAuthor = 1 - printAuthor;
 185                         break;
 186                 case 'd':
 187                         printDate = 1 - printDate;
 188                         break;
 189                 case 'v':
 190                         printVersion = 1 - printDate;
 191                         break;
 192                 case 'h':
 193                         fprintf(ostate->output, "%s", usage);
 194                         return 0;
 195                 default:
 196                         fprintf(ostate->output, "ERROR: Unknown short option '%c'\n", opt);
 197                         fprintf(ostate->output, "%s", usage);
 198                         return 1;
 199                 }
 200         }
 201 
 202         /* Print version info. */
 203         if(printVersion) fprintf(ostate->output, "\tTechOS v%d.%d\n", major_ver, minor_ver);
 204         if(printAuthor)  fprintf(ostate->output, "\tAuthors: Benjamin Culkin, Lucas Darnell, Jared Miller\n");
 205         if(printDate)    fprintf(ostate->output, "\tCompletion Date: 9/13/17\n");
 206

 210 HANDLECOM(help) {
 211         /* Usage message. */
 212         char *usage = "Usage: help [-h] [--help] [<command-name>]\n";
 213 
 214         /* Handle CLI args. */
 215         if(argc == 1) {
 216                 fprintf(ostate->output, "Available commands:\n");
 217 
 218                 /* 
 219                  * @TODO 10/07/17 Ben Culkin :PrintCommandState
 220                  *       printcommands needs to be rewritten to accept a
 221                  *       commandstate, so that it can handle things like command
 222                  *       categories and help-only commands/internal commands
 223                  *       that don't show up in the default help list.
 224                  */
 225                 printcommands(ostate->pComstate->plCommands, ostate->output);
 226         } else if(argc == 2){
 227                 /* The path to the man page. */
 228                 char *manpath;
 229 
 230                 /* The struct for checking if a file exists. */
 231                 struct stat scratch;
 232 
 233                 int ret;
 234 
 235                 /* Handle CLI args. */
 236                 if(argv[1][0] == '-') {
 237                         checkhelpargs(argc, argv, usage, ostate);
 238                         return 1;
 239                 }
 240 
 241                 /* Get the path we want to use. */
 242                 assert(asprintf(&manpath, "help/%s.1", argv[1]) != -1);
 243 
 244                 /* Check if the file exists. */
 245                 if(stat(manpath, &scratch) == -1) {
 246                         /* File wasn't found. */
 247                         fprintf(ostate->output, "\tINTERNAL ERROR: Manpage for command '%s' ('%s') couldn't be found.\n", argv[1], manpath);
 248 
 249                         free(manpath);
 250                         return 1;
 251                 }
 252                 if(S_ISREG(scratch.st_mode)) {
 253                         /* Command to run. */
 254                         char *compath;
 255 
 256                         /* Create our command, then run it. */
 257                         assert(asprintf(&compath, "man %s", manpath) != -1);
 258                         if(system(compath) != 0) {
 259                                 fprintf(ostate->output, "\tINTERNAL ERROR: Could not display the manpage for '%s'\n", argv[1]);
 260                         }
 261 
 262                         free(compath);  
 263                 } else {
 264                         fprintf(ostate->output, "\tINTERNAL ERROR: Manpage for command '%s' ('%s') is not valid.\n", argv[1], manpath);
 265 
 266                         free(manpath);
 267                         return 1;
 268                 }
 269 
 270                 free(manpath);
 271         } else {
 272                 fprintf(ostate->output, "\tERROR: Invalid command-line arguments\n");
 273                 fprintf(ostate->output, "\t%s", usage);
 274         }
 275 
 276         return 0;
 277 }
